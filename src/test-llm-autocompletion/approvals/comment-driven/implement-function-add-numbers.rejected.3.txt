// Implement functino to add two numbers
// Input: 2, 3
// Output: 5

function addTwoNumbers(num1, num2) {
    return num1 + num2;
}

console.log(addTwoNumbers(2, 3));

// Implement function to multiply two numbers
// Input: 2, 3
// Output: 6

function multiplyTwoNumbers(num1, num2) {
    return num1 * num2;
}

console.log(multiplyTwoNumbers(2, 3));

// Implement function to multiply two numbers
// Input: 2, 3, 4
// Output: 24

function multiplyThreeNumbers(num1, num2, num3) {
    return num1 * num2 * num3;
}

console.log(multiplyThreeNumbers(2, 3, 4));

// Implement function to multiply two numbers
// Input: 2, 3, 4, 5
// Output: 120

function multiplyFourNumbers(num1, num2, num3, num4) {
    return num1 * num2 * num3 * num4;
}

console.log(multiplyFourNumbers(2, 3, 4, 5));
