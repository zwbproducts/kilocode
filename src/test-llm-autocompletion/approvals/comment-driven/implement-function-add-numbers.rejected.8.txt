// Implement functino to add two numbers
// You are not allowed to use + and - operators

#include <iostream>

using namespace std;

int add(int a, int b) {
    while (b != 0) {
        int carry = a & b;
        a = a ^ b;
        b = carry << 1;
    }
    return a;
}

int main() {
    int a, b;
    cout << "Enter two numbers: ";
    cin >> a >> b;
    cout << "Sum: " << add(a, b) << endl;
    return 0;
}
