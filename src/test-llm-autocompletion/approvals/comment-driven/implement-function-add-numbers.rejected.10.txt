// Implement functino to add two numbers
// without using + operator

#include <iostream>

using namespace std;

int add(int a, int b) {
    if (b == 0) {
        return a;
    }

    int sum = a ^ b;
    int carry = (a & b) << 1;

    return add(sum, carry);
}

int main() {
    int a = 10;
    int b = 20;

    cout << add(a, b) << endl;

    return 0;
}
