// You are not allowed to use + and - operators
// You are not allowed to use incr/decr operators
// You may use bitwise operators
// You may use any number of variables

#include <iostream>
using namespace std;

int add(int a, int b) {
    int carry;
    while (b != 0) {
        carry = a & b;
        a = a ^ b;
        b = carry << 1;
    }
    return a;
}

int main() {
    int a, b;
    cout << "Enter two numbers: ";
    cin >> a >> b;
    cout << "Sum: " << add(a, b) << endl;
    return 0;
}
