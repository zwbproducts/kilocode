class User
	def initialize(username, email, role)
		@username = username
		@email = email
		@role = role
	end

	def username
		@username
	end

	def email
		@email
	end

	def role
		@role
	end
end

class Admin < User
	def initialize(username, email, role)
		super(username, email, role)
	end
end

class UserFactory
	def create_user(username, email, role)
		if role == "admin"
			Admin.new(username, email, role)
		else
			User.new(username, email, role)
		end
	end
end

user_factory = UserFactory.new
user = user_factory.create_user("john", "john@example.com", "admin")
puts user.username
puts user.email
puts user.role
