def calculate_sum(a, b)
	sum = a + b
	return sum
end

def calculate_difference(a, b)
	difference = a - b
	return difference
end

def calculate_product(a, b)
	product = a * b
	return product
end

def calculate_quotient(a, b)
	quotient = a / b
	return quotient
end

def calculate_remainder(a, b)
	remainder = a % b
	return remainder
end

def calculate_exponent(a, b)
	exponent = a ** b
	return exponent
end
